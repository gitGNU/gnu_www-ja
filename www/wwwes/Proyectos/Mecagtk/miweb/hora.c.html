#include <stdio.h>
#include <assert.h>
#include "hora.h"

void HoraInicializa(Hora *pH, int hh, int mm, int ss)
{
    assert ( pH != NULL );
    assert ( 0 <= hh && hh < 24 );
    assert ( 0 <= mm && mm < 60 );
    assert ( 0 <= ss && ss < 60 );
    pH->hh = hh;
    pH->mm = mm;
    pH->ss = ss;
}

char *HoraACadena( Hora *pH)
{
    static char buffer[9];
    sprintf(buffer,"%02d:%02d:%02d",pH->hh,pH->mm,pH->ss);
    return buffer;
}
