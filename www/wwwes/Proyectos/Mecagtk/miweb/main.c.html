#include <stdio.h>
#include "matriz.h"
#include "hora.h"

main()
{
    Matriz A,B;
    int m,n,i,j;
    float x;
    Hora h;
    
    /* Obtener la dimensión de la matriz */
    printf("Altura  :");
    scanf("%d",&m);
    printf("Anchura :");
    scanf("%d",&n);

    /* Creamos una matriz de floats */
    MatrizCrea(&A, m, n, sizeof(float));

    /* Creamos una matriz de horas */
    MatrizCrea(&B, m, n, sizeof(h));

    /* Inicializamos las Matrices */
    for(i=0;i < m; i++)
        for(j=0;j < n; j++)
            {
                x=(i+j+2.0)/(n+m);
                MatrizCambia(&A,i,j,&x);
                HoraInicializa(&h,1,i,j);
                MatrizCambia(&B,i,j,&h);
            }

    /* Recorremos la matriz de floats */
    printf("\nMatriz de FLOATS ((i+j+2.0)/(alt+anch)):\n"); 
    for(i=0;i < m; i++)
    {
        for(j=0;j < n; j++)
            {
                MatrizObten(&A,i,j,&x);
                printf("%f ",x);
            }      
        printf("\n");
    }

    /* Recorremos la matriz de Hora */
     printf("\nMatriz de Horas:\n"); 
    for(i=0;i < m; i++)
    {
        for(j=0;j < n; j++)
            {
                MatrizObten(&B,i,j,&h);
                printf("%s ",HoraACadena(&h));
            }
        printf("\n");
    }         


    /* Limpieza */
    MatrizDestruye(&A);
    MatrizDestruye(&B);
}
