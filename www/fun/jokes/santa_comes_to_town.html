Santa Claus is coming to town

#!/bin/santash  

# Santa Claus is coming to town

better !pout !cry
better watchout
lpr why
santa_claus < north_pole >town

cat /etc/passwd > list
ncheck list
ncheck list
cat list | grep naughty >coal
cat list | grep nice >gift
santa_claus < north_pole >town

who | grep sleeping
who | grep awake
who | egrep 'bad|good'

for goodness_sake; do
  be_good;
done
